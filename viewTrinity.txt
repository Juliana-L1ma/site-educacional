CREATE VIEW turmasPorCurso AS
SELECT
    c.nome_curso AS NomeDoCurso,
    t.nome_turma AS NomeDaTurma
FROM
    cursos c
JOIN
    turmas t ON c.id_curso = t.id_curso;







CREATE VIEW disciplinasPorCurso AS
SELECT
    c.nome_curso AS NomeDoCurso,
    u.nome_uc AS NomeDaDisciplina
FROM
    cursos c
JOIN
    unidades_curriculares u ON c.id_unidade_curricular = u.id_unid_curricular;
